- 借助数组
  - O(n)
  - O(n)
```go
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func isPalindrome(head *ListNode) bool {
    ans := []int{}
    for head != nil{
        ans = append(ans , head.Val)
        head = head.Next
    }
    for i := 0 ; i < len(ans) / 2 ; i++{
        if ans[i] != ans[len(ans) - 1 - i]{
            return false
        }
    }
    return true
}
```
